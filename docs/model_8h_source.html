92  // processes a node in a recursive fashion. Processes each individual mesh located at the node and repeats this process on its children nodes (if any).
93  void processNode(aiNode *node, const aiScene *scene)
94  {
95  // process each mesh located at the current node
96  for(unsigned int i = 0; i < node->mNumMeshes; i++)
97  {
98  // the node object only contains indices to index the actual objects in the scene.
99  // the scene contains all the data, node is just to keep stuff organized (like relations between nodes).
100  aiMesh* mesh = scene->mMeshes[node->mMeshes[i]];
101  meshes.push_back(processMesh(mesh, scene));
102  }
103  // after we've processed all of the meshes (if any) we then recursively process each of the children nodes
104  for(unsigned int i = 0; i < node->mNumChildren; i++)
105  {
106  processNode(node->mChildren[i], scene);
107  }

112  Mesh processMesh(aiMesh *mesh, const aiScene *scene)
113  {
114  vector<Vertex> vertices;
115  vector<unsigned int> indices;
116  vector<Texture> textures;
117 
118  // Walk through each of the mesh's vertices
119  for(unsigned int i = 0; i < mesh->mNumVertices; i++)
120  {
121  Vertex vertex;
122  glm::vec3 vector; // we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
123  // positions
124  vector.x = mesh->mVertices[i].x;
125  vector.y = mesh->mVertices[i].y;
126  vector.z = mesh->mVertices[i].z;
127  vertex.Position = vector;
128 
129  // texture coordinates
130  if(mesh->mTextureCoords[0]) // does the mesh contain texture coordinates?
131  {
132  glm::vec2 vec;
133  // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
134  // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
135  vec.x = mesh->mTextureCoords[0][i].x;
136  vec.y = mesh->mTextureCoords[0][i].y;
137  vertex.TexCoords = vec;
138  }
139  else
140  vertex.TexCoords = glm::vec2(0.0f, 0.0f);
141 
142  vertices.push_back(vertex);
143  }
144  // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
145  for(unsigned int i = 0; i < mesh->mNumFaces; i++)
146  {
147  aiFace face = mesh->mFaces[i];
148  // retrieve all indices of the face and store them in the indices vector
149  for(unsigned int j = 0; j < face.mNumIndices; j++)
150  indices.push_back(face.mIndices[j]);
151  }
152  // process materials
153  aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
154  // we assume a convention for sampler names in the shaders. Each diffuse texture should be named
155  // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
156  // Same applies to other texture as the following list summarizes:
157  // diffuse: texture_diffuseN
158  // specular: texture_specularN
159  // normal: texture_normalN
160 
161  // 1. diffuse maps
162  vector<Texture> diffuseMaps = loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
163  textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
164  // 2. specular maps
165  vector<Texture> specularMaps = loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
166  textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
167  // 3. normal maps
168  std::vector<Texture> normalMaps = loadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
169  textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
170  // 4. height maps
171  std::vector<Texture> heightMaps = loadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
172  textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
173 
174  // return a mesh object created from the extracted mesh data
175  return Mesh(vertices, indices, textures);
176  }

178  // checks all material textures of a given type and loads the textures if they're not loaded yet.
179  // the required info is returned as a Texture struct.
180  vector<Texture> loadMaterialTextures(aiMaterial *mat, aiTextureType type, string typeName)
181  {
182  vector<Texture> textures;
183  for(unsigned int i = 0; i < mat->GetTextureCount(type); i++)
184  {
185  aiString str;
186  mat->GetTexture(type, i, &str);
187  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
188  bool skip = false;
189  for(unsigned int j = 0; j < textures_loaded.size(); j++)
190  {
191  if(std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
192  {
193  textures.push_back(textures_loaded[j]);
194  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
195  break;
196  }
197  }
198  if(!skip)
199  { // if texture hasn't been loaded already, load it
200  Texture texture;
201  texture.id = TextureFromFile(str.C_Str(), this->directory);
202  texture.type = typeName;
203  texture.path = str.C_Str();
204  textures.push_back(texture);
205  textures_loaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
206  }
207  }
208  return textures;
209  }

213 unsigned int TextureFromFile(const char *path, const string &directory, bool gamma)
214 {
215  string filename = string(path);
216  filename = directory + '/' + filename;
217 
218  unsigned int textureID;
219  glGenTextures(1, &textureID);
220  stbi_set_flip_vertically_on_load(true); // tell stb_image.h to flip loaded texture's on the y-axis.
221 
222  int width, height, nrComponents;
223  unsigned char *data = stbi_load(filename.c_str(), &width, &height, &nrComponents, 0);
224  if (data)
225  {
226  GLenum format;
227  if (nrComponents == 1)
228  format = GL_RED;
229  else if (nrComponents == 3)
230  format = GL_RGB;
231  else if (nrComponents == 4)
232  format = GL_RGBA;
233 
234  glBindTexture(GL_TEXTURE_2D, textureID);
235  glTexImage2D(GL_TEXTURE_2D, 0, format, width, height, 0, format, GL_UNSIGNED_BYTE, data);
236  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
237  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
238  // set texture filtering parameters
239  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
240  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
241 
242  stbi_image_free(data);
243  }
244  else
245  {
246  std::cout << "Texture failed to load at path: " << path << std::endl;
247  stbi_image_free(data);
248  }
249 
250  return textureID;
251 }
