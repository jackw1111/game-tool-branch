47  // render the mesh
48  void Draw(Shader &shader)
49  {
50  // bind appropriate textures
51  unsigned int diffuseNr = 1;
52  unsigned int specularNr = 1;
53  unsigned int normalNr = 1;
54  unsigned int heightNr = 1;
55  for(unsigned int i = 0; i < textures.size(); i++)
56  {
57  glActiveTexture(GL_TEXTURE0 + i); // active proper texture unit before binding
58  // retrieve texture number (the N in diffuse_textureN)
59  string number;
60  string name = textures[i].type;
61  if(name == "texture_diffuse")
62  number = std::to_string(diffuseNr++);
63  else if(name == "texture_specular")
64  number = std::to_string(specularNr++); // transfer unsigned int to stream
65  else if(name == "texture_normal")
66  number = std::to_string(normalNr++); // transfer unsigned int to stream
67  else if(name == "texture_height")
68  number = std::to_string(heightNr++); // transfer unsigned int to stream
69 
70  // now set the sampler to the correct texture unit
71  glUniform1i(glGetUniformLocation(shader.ID, (name + number).c_str()), i);
72  // and finally bind the texture
73  glBindTexture(GL_TEXTURE_2D, textures[i].id);
74  }
75 
76  // draw mesh
77  glBindVertexArray(VAO);
78  glDrawElements(GL_TRIANGLES, indices.size(), GL_UNSIGNED_INT, 0);
79  glBindVertexArray(0);
80 
81  // always good practice to set everything back to defaults once configured.
82  glActiveTexture(GL_TEXTURE0);
83  }

89  // initializes all the buffer objects/arrays
90  void setupMesh()
91  {
92  // create buffers/arrays
93  glGenVertexArrays(1, &VAO);
94  glGenBuffers(1, &VBO);
95  glGenBuffers(1, &EBO);
96 
97 
98 
99  glBindVertexArray(VAO);
100  // load data into vertex buffers
101  glBindBuffer(GL_ARRAY_BUFFER, VBO);
102  // A great thing about structs is that their memory layout is sequential for all its items.
103  // The effect is that we can simply pass a pointer to the struct and it translates perfectly to a glm::vec3/2 array which
104  // again translates to 3/2 floats which translates to a byte array.
105  glBufferData(GL_ARRAY_BUFFER, vertices.size() * sizeof(Vertex), &vertices[0], GL_STATIC_DRAW);
106 
107  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, EBO);
108  glBufferData(GL_ELEMENT_ARRAY_BUFFER, indices.size() * sizeof(unsigned int), &indices[0], GL_STATIC_DRAW);
109 
110  // set the vertex attribute pointers
111  // vertex Positions
112  glEnableVertexAttribArray(0);
113  glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)0);
114  // vertex texture coords
115  glEnableVertexAttribArray(1);
116  glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(Vertex), (void*)offsetof(Vertex, TexCoords));
117 
118  glBindVertexArray(0);
119  }
